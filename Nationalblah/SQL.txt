CREATE TABLE University (
id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
name VARCHAR(256),
fee INTEGER(6),
total_enrollment INTEGER(6),
scores_id INTEGER,
rank_id INTEGER
)


CREATE TABLE Scores(
id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
sat INTEGER(5),
gpa FLOAT(4)
)

CREATE TABLE Rank(
id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
qs INTEGER,
sh INTEGER
)
